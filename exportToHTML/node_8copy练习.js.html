//fs操作 
var fs = require('fs'); 
function copy(src,target){ 
    fs.readFile(src,function(err,data){ 
 
        fs.writeFile(target,data,function(){ 
            if(err){ 
                console.error(err); 
            } 
            console.log('copy from '+src+' is successful to '+target) 
        }) 
    }) 
 
} 
copy('1.jpg','2.jpg');
